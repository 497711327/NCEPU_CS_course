#include<stdio.h>
#define N 80
void count(char str[],int cou[])
{
	
	int i;
	for(i=0;str[i];i++)
	{
		if(str[i]>='A'&&str[i]<='Z'||str[i]>='a'&&str[i]<='z')
			cou[0]++;
		else if(str[i]>='0'&&str[i]<='9')
			cou[1]++;
		else
			cou[2]++;
	}
	
}
